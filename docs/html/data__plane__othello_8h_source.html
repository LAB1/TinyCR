68  if (VDL == 0) return;

71

73  uint8_t offset = uint8_t(index * VDL % 64);
74  char left = char(offset + VDL - 64);

196  inline bool query(const K &k, V &v) const {
197  uint32_t ha, hb;
198  getIndices(k, ha, hb);
199  uint64_t aa = memGet(ha);
200  uint64_t bb = memGet(hb);
202  uint64_t vd = aa ^bb;
203 
204  v = vd >> DL; // extract correct v
205 
206  if (DL == 0) return true; // no filter features
207 
208  if ((aa & 1) == 0 || (bb & 1) == 0) return false; // with filter features, then the last bit must be 1
209 
210  if (DL == 1) return true; // shortcut for one bit digest
211 
212  uint32_t digest = uint32_t(vd & DEMASK);
213  return (digest | 1) == ((hd(k) & DEMASK) | 1); // ignore the last bit
214  }
