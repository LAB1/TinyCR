50  inline void memSet(uint32_t index, uint64_t value) {
51  uint64_t v = uint64_t(value) & MASK;
52 
53  uint32_t start = index * L / 64;
54  uint8_t offset = uint8_t(index * L % 64);
55  char left = char(offset + L - 64);
56 
57  uint64_t mask = ~(MASK << offset); // [offset, offset + L) should be 0, and others are 1
58 
59  mem[start] &= mask;
60  mem[start] |= v << offset;
61 
62  if (L > 1 && left > 0) {
63  mask = uint64_t(-1) << left; // lower left bits should be 0, and others are 1
64  mem[start + 1] &= mask;
65  mem[start + 1] |= v >> (L - left);
66  }
67  }

69  inline uint64_t getAt(uint32_t index) const {
70  uint32_t start = index * L / 64;
71  uint8_t offset = uint8_t(index * L % 64);
72 
73  char left = char(L > 1 ? char(offset + L - 64) : 0);
74  left = char(left < 0 ? 0 : left);
75 
76  uint64_t mask = ~(uint64_t(-1) << (L - left)); // lower L-left bits should be 1, and others are 0
77  uint64_t result = (mem[start] >> offset) & mask;
78 
79  if (L > 1 && left > 0) {
80  mask = ~(uint64_t(-1) << left); // lower left bits should be 1, and others are 0
81  result |= (mem[start + 1] & mask) << (L - left);
82  }
