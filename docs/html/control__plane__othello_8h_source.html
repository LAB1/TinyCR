109  inline void memSet(uint32_t index, uint64_t value) {
110  if (VDL == 0) return;
111 
112  uint64_t v = uint64_t(value) & VDEMASK;
113 
114  uint32_t start = index * VDL / 64;
115  uint8_t offset = uint8_t(index * VDL % 64);
116  char left = char(offset + VDL - 64);
117 
118  uint64_t mask = ~(VDEMASK << offset); // [offset, offset + VDL) should be 0, and others are 1
119 
120  mem[start] &= mask;
121  mem[start] |= v << offset;
122 
123  if (left > 0) {
124  mask = uint64_t(-1) << left; // lower left bits should be 0, and others are 1
125  mem[start + 1] &= mask;
126  mem[start + 1] |= v >> (VDL - left);
127  }
128 
129  }

133  inline uint64_t memGet(uint32_t index) const {
134  if (VDL == 0) return 0;
135 
136  uint32_t start = index * VDL / 64;
137  uint8_t offset = uint8_t(index * VDL % 64);
138 
139  char left = char(offset + VDL - 64);
140  left = char(left < 0 ? 0 : left);
141 
142  uint64_t mask = ~(uint64_t(-1) << (VDL - left)); // lower VDL-left bits should be 1, and others are 0
143  uint64_t result = (mem[start] >> offset) & mask;
144 
145  if (left > 0) {
146  mask = ~(uint64_t(-1) << left); // lower left bits should be 1, and others are 0
147  result |= (mem[start + 1] & mask) << (VDL - left);
148  }
149 
150  return result;
151  }

501  if (fillValue && maintainingDP) {

504  uint64_t v = values[keyId];

513  uint64_t v = values[keyId];
514  valueToFill = v ^ memGet(oppositeNode);
515  }
516 
517  memSet(nodeToFill, valueToFill);
518  }
519  }
520

522  uint32_t indexToFill = keyId ^indMem[oppositeNode];
