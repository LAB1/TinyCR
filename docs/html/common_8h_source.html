116 template<class IntType = unsigned long, class RealType = double>
117 class zipf_distribution {
118 public:
119  typedef RealType input_type;
120  typedef IntType result_type;
121 
122  static_assert(std::numeric_limits<IntType>::is_integer, "");
123  static_assert(!std::numeric_limits<RealType>::is_integer, "");
124 
125  zipf_distribution(const IntType n = std::numeric_limits<IntType>::max(),
126  const RealType q = 1.0)
127  : n(n), q(q), H_x1(H(1.5) - 1.0), H_n(H(n + 0.5)), dist(H_x1, H_n) {}
128 
129  IntType operator()(std::default_random_engine &rng) {
130  while (true) {
131  const RealType u = dist(rng);
132  const RealType x = H_inv(u);
133  const IntType k = clamp<IntType>(std::round(x), 1, n);
134  if (u >= H(k + 0.5) - h(k)) {
135  return k;
136  }
137  }
138  }
139 
140 private:
142  template<typename T>
143  static constexpr T clamp(const T x, const T min, const T max) {
144  return std::max(min, std::min(max, x));
145  }
146 
148  static double
149  expxm1bx(const double x) {
150  return (std::abs(x) > epsilon)
151  ? std::expm1(x) / x
152  : (1.0 + x / 2.0 * (1.0 + x / 3.0 * (1.0 + x / 4.0)));
153  }
154 
161  const RealType H(const RealType x) {
162  const RealType log_x = std::log(x);
163  return expxm1bx((1.0 - q) * log_x) * log_x;
164  }
165 
167  static RealType
168  log1pxbx(const RealType x) {
169  return (std::abs(x) > epsilon)
170  ? std::log1p(x) / x
171  : 1.0 - x * ((1 / 2.0) - x * ((1 / 3.0) - x * (1 / 4.0)));
172  }
173 
175  const RealType H_inv(const RealType x) {
176  const RealType t = std::max(-1.0, x * (1.0 - q));
177  return std::exp(log1pxbx(t) * x);
178  }
179 
181  const RealType h(const RealType x) {
182  return std::exp(-q * std::log(x));
183  }
184 
185  static constexpr RealType epsilon = 1e-8;
186 
187  IntType n;
188  RealType q;
189  RealType H_x1;
190  RealType H_n;
191  std::uniform_real_distribution<RealType> dist;
192 };
